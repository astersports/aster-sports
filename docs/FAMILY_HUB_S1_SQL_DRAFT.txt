SCHEMA FACTS THAT SHAPED THE SQL (verified, not inferred):
- guardians.email is NULLABLE (4 NULL rows today) -> match predicate requires
  guardians.email IS NOT NULL so NULL never matches.
- guardians.email UNIQUE (guardians_email_unique, single-col) -> at most one
  guardian per email today; C1 multi-match loop is future-proofing for org #2.
- guardians.user_id FK->auth.users(id), nullable -> NULL->self is the only write.
- user_roles UNIQUE(user_id, organization_id); role CHECK in {admin,coach,parent}
  -> ON CONFLICT (user_id, organization_id) DO NOTHING valid; 'parent' passes.
- players: id, org_id NOT NULL, first_name, last_name, grade, member_type.
- auth.users.email varchar, email_confirmed_at timestamptz -> read both server-side.
- current_user_player_ids() exists (SECDEF STABLE, uuid[]) -> reused by the S2
  INVOKER read intersection (documented, not built in S1).
- guardians_update_own USING = user_id = (SELECT auth.uid()) -> confirms the gap:
  an unclaimed row (user_id IS NULL) fails this -> client UPDATE is RLS-blocked
  -> SECDEF RPC required.
- No family_*/child_links/calendar_subscriptions/synced_events tables -> clean build.

-- READ-PATH BOUNDARY (S2, documented NOT built): the Hub READ composition is
-- SECURITY INVOKER (runs as the parent), unioning (a) NATIVE events through the
-- existing org RLS chain intersected with current_user_player_ids() so a forged
-- child_links row can NEVER widen native visibility, and (b) synced_events from
-- the caller's own subscriptions. NO SECDEF over the read path. SECDEF is used
-- ONLY for the claim/populate WRITES below. (C2 COND-1.)

-- ---- PART 2: populate_household() — C2 COND-2 ------------------------------
-- SECDEF, scoped strictly to auth.uid(), NO target-user param. Writes only
-- owner_user_id = auth.uid() rows. Idempotent (ON CONFLICT no-op).
CREATE OR REPLACE FUNCTION public.populate_household()
RETURNS jsonb LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, auth AS $$
DECLARE
  v_uid uuid := auth.uid();
  v_created_kids int := 0; v_created_links int := 0;
  r record; v_child_id uuid;
BEGIN
  IF v_uid IS NULL THEN RETURN jsonb_build_object('ok', false, 'reason', 'unauthenticated'); END IF;
  FOR r IN
    SELECT DISTINCT p.id AS player_id, p.org_id, p.first_name, p.last_name, p.grade
    FROM public.guardians g
    JOIN public.player_guardians pg ON pg.guardian_id = g.id
    JOIN public.players p ON p.id = pg.player_id
    WHERE g.user_id = v_uid
  LOOP
    SELECT cl.child_id INTO v_child_id FROM public.child_links cl
     WHERE cl.owner_user_id = v_uid AND cl.link_type = 'ember_player'
       AND cl.ember_org_id = r.org_id AND cl.ember_player_id = r.player_id LIMIT 1;
    IF v_child_id IS NULL THEN
      INSERT INTO public.family_children (owner_user_id, display_name, grade)
      VALUES (v_uid, trim(r.first_name || ' ' || r.last_name), r.grade)
      RETURNING id INTO v_child_id;
      v_created_kids := v_created_kids + 1;
      INSERT INTO public.child_links (owner_user_id, child_id, link_type, ember_org_id, ember_player_id)
      VALUES (v_uid, v_child_id, 'ember_player', r.org_id, r.player_id)
      ON CONFLICT (owner_user_id, ember_org_id, ember_player_id) WHERE link_type = 'ember_player' DO NOTHING;
      v_created_links := v_created_links + 1;
    END IF;
  END LOOP;
  RETURN jsonb_build_object('ok', true, 'children_created', v_created_kids, 'links_created', v_created_links);
END; $$;
REVOKE EXECUTE ON FUNCTION public.populate_household() FROM PUBLIC;
REVOKE EXECUTE ON FUNCTION public.populate_household() FROM anon;
GRANT  EXECUTE ON FUNCTION public.populate_household() TO authenticated;

-- ---- PART 1: claim_guardian() — C1, all 8 conditions -----------------------
-- NO org param (Q1): claims ALL unclaimed rows matching the confirmed session
-- email (COND-5 cross-org). Inserts user_roles(parent) per claimed org.
CREATE OR REPLACE FUNCTION public.claim_guardian()
RETURNS jsonb LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, auth AS $$
DECLARE
  v_uid uuid := auth.uid();
  v_email text; v_confirmed timestamptz; v_norm text;
  v_claimed_ids uuid[] := '{}'; g record;
BEGIN
  IF v_uid IS NULL THEN RETURN jsonb_build_object('claimed', false, 'reason', 'unauthenticated'); END IF;
  -- C1.1 session-derived + C1.2 confirmed-only
  SELECT u.email::text, u.email_confirmed_at INTO v_email, v_confirmed FROM auth.users u WHERE u.id = v_uid;
  IF v_confirmed IS NULL THEN RETURN jsonb_build_object('claimed', false, 'reason', 'email_not_confirmed'); END IF;
  v_norm := lower(trim(v_email));
  IF v_norm IS NULL OR v_norm = '' THEN RETURN jsonb_build_object('claimed', false, 'reason', 'no_session_email'); END IF;
  -- C1.3 exact match + C1.4 NULL->self + C1.5 multi-match (skip claimed via user_id IS NULL)
  FOR g IN
    SELECT gd.id, gd.org_id FROM public.guardians gd
     WHERE gd.user_id IS NULL AND gd.email IS NOT NULL AND lower(trim(gd.email)) = v_norm
     FOR UPDATE
  LOOP
    UPDATE public.guardians SET user_id = v_uid, updated_at = now() WHERE id = g.id AND user_id IS NULL;
    IF FOUND THEN
      v_claimed_ids := array_append(v_claimed_ids, g.id);
      INSERT INTO public.user_roles (user_id, organization_id, role)
      VALUES (v_uid, g.org_id, 'parent') ON CONFLICT (user_id, organization_id) DO NOTHING;
    END IF;
  END LOOP;
  -- C1.7 atomic: populate from whatever the caller now guards (backfill on re-claim too)
  PERFORM public.populate_household();
  -- C1.8 fail closed
  IF array_length(v_claimed_ids, 1) IS NULL THEN
    IF EXISTS (SELECT 1 FROM public.guardians WHERE user_id = v_uid) THEN
      RETURN jsonb_build_object('claimed', false, 'reason', 'already_claimed', 'guardian_ids', '[]'::jsonb);
    END IF;
    RETURN jsonb_build_object('claimed', false, 'reason', 'no_matching_guardian');
  END IF;
  RETURN jsonb_build_object('claimed', true, 'guardian_ids', to_jsonb(v_claimed_ids),
    'orgs_claimed', (SELECT count(DISTINCT organization_id) FROM public.user_roles WHERE user_id = v_uid));
END; $$;
REVOKE EXECUTE ON FUNCTION public.claim_guardian() FROM PUBLIC;
REVOKE EXECUTE ON FUNCTION public.claim_guardian() FROM anon;
GRANT  EXECUTE ON FUNCTION public.claim_guardian() TO authenticated;

APPENDIX A — CONDITION-COMPLIANCE CHECKLIST
==============================================================================
C1 claim_guardian:
 1 session-derived email: no params; SELECT u.email FROM auth.users WHERE id=v_uid
 2 confirmed-only: IF v_confirmed IS NULL -> 'email_not_confirmed'
 3 exact normalized match: lower(trim(gd.email)) = v_norm  (no ILIKE)
 4 NULL->self: WHERE gd.user_id IS NULL loop + UPDATE ... WHERE user_id IS NULL re-check
 5 multi-match skip-claimed: loop all unclaimed matches; claimed excluded by user_id IS NULL
 6 SECDEF hardening: SECURITY DEFINER, SET search_path=public,auth, REVOKE PUBLIC+anon, GRANT authenticated
 7 atomic+idempotent: one body (flip + user_roles ON CONFLICT + populate); re-check + ON CONFLICT = no-op
 8 fail-closed: each guard RETURNs structured {claimed:false} and writes nothing
 user_roles parent row: YES (required — parent needs BOTH user_id AND user_roles). Inserted per claimed org.
C2:
 COND-1 read INVOKER, no SECDEF over reads: no read fn in S1; READ-PATH BOUNDARY comment documents S2 INVOKER + current_user_player_ids() intersection
 COND-2 populate scoped to auth.uid(): no param; v_uid:=auth.uid(); writes only owner_user_id=v_uid; ON CONFLICT idempotent
